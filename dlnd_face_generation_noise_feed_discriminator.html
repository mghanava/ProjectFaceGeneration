Face Generation¶

In this project, you'll define and train a DCGAN on a dataset of faces. Your goal is to get a generator network to generate new images of faces that look as realistic as possible!

The project will be broken down into a series of tasks from loading in data to defining and training adversarial networks. At the end of the notebook, you'll be able to visualize the results of your trained Generator to see how it performs; your generated samples should look like fairly realistic faces with small amounts of noise.

class Discriminator(nn.Module):

    def __init__(self, conv_dim=32):
        """
        Initialize the Discriminator Module
        :param conv_dim: The depth of the first convolutional layer
        """
        super(Discriminator, self).__init__()

        # complete init function
        self.conv_dim = conv_dim
        
        ## Define downsampler
        self.features = nn.Sequential(collections.OrderedDict([
            ('conv1', nn.Conv2d(in_channels=3, out_channels=conv_dim*4, 
                                kernel_size=4, stride=2, padding=1, bias=True)),
            ('leakyrelu1', nn.LeakyReLU(negative_slope=0.2, inplace=True)),         
            ('conv2', nn.Conv2d(in_channels=conv_dim*4, out_channels=conv_dim*8, 
                                kernel_size=4, stride=2, padding=1, bias=False)),
            ('batchnorm1', nn.BatchNorm2d(num_features=conv_dim*8)),
            ('leakyrelu2', nn.LeakyReLU(negative_slope=0.2, inplace=True)),
            ('conv3', nn.Conv2d(in_channels=conv_dim*8, out_channels=conv_dim*16, 
                                kernel_size=4, stride=2, padding=1, bias=False)),
            ('batchnorm2', nn.BatchNorm2d(num_features=conv_dim*16)),
            ('leakyrelu3', nn.LeakyReLU(negative_slope=0.2, inplace=True)) 
        ]))
        
        self.classifier = nn.Sequential(collections.OrderedDict([           
            ('fc', nn.Linear(in_features=conv_dim*16*4*4, out_features=1, bias=True))
        ]))
      
   
    def forward(self, x):
        """
        Forward propagation of the neural network
        :param x: The input to the neural network     
        :return: Discriminator logits; the output of the neural network
        """
        # define feedforward behavior
        
        x = self.features(x)
        x = x.view(-1, self.conv_dim*16*4*4) # flatten feature vector prior to entering classifier
        x = self.classifier(x)

class Generator(nn.Module):
    
    def __init__(self, z_size, conv_dim=32):
        """
        Initialize the Generator Module
        :param z_size: The length of the input latent vector, z
        :param conv_dim: The depth of the inputs to the *last* transpose convolutional layer
        """
        super(Generator, self).__init__()

        # complete init function
        
        self.conv_dim = conv_dim

        ## Define layers of a CNN
        self.vectorizer = nn.Sequential(collections.OrderedDict([           
            ('fc', nn.Linear(in_features=z_size, out_features=conv_dim*16*4*4, bias=True))
        ]))
        
        self.upsampler = nn.Sequential(collections.OrderedDict([
            ('transpose_conv1', nn.ConvTranspose2d(in_channels=conv_dim*16, out_channels=conv_dim*8, 
                                kernel_size=4, stride=2, padding=1, bias=False)),
            ('batchnorm1', nn.BatchNorm2d(num_features=conv_dim*8)),
            ('relu1', nn.ReLU(inplace=True)),         
            ('transpose_conv2', nn.ConvTranspose2d(in_channels=conv_dim*8, out_channels=conv_dim*4, 
                                kernel_size=4, stride=2, padding=1, bias=False)),
            ('batchnorm2', nn.BatchNorm2d(num_features=conv_dim*4)),
            ('relu2', nn.ReLU(inplace=True)),
            ('transpose_conv3', nn.ConvTranspose2d(in_channels=conv_dim*4, out_channels=3, 
                                kernel_size=4, stride=2, padding=1, bias=True)),
            ('tanh', nn.Tanh())  
        ]))
        
    def forward(self, x):
        """
        Forward propagation of the neural network
        :param x: The input to the neural network     
        :return: A 32x32x3 Tensor image as output
        """
        # define feedforward behavior
        x = self.vectorizer(x)
        x = x.view(-1, self.conv_dim*16, 4, 4) 
        x = self.upsampler(x)
        
        return x

Discriminator and Generator Losses¶

Now we need to calculate the losses for both types of adversarial networks.

Discriminator Losses¶

  • For the discriminator, the total loss is the sum of the losses for real and fake images, d_loss = d_real_loss + d_fake_loss.
  • Remember that we want the discriminator to output 1 for real images and 0 for fake images, so we need to set up the losses to reflect that.

Generator Loss¶

The generator loss will look similar only with flipped labels. The generator's goal is to get the discriminator to think its generated images are real.

You may choose to use either cross entropy or a least squares error loss to complete the following real_loss and fake_loss functions.

def real_loss(D_out, smooth=False):
    '''Calculates how close discriminator outputs are to being real.
       param, D_out: discriminator logits
       return: real loss'''
    batch_size = D_out.size(0)
    if smooth:
        labels = torch.ones(batch_size) * 0.9
    else:
        labels = torch.ones(batch_size)
    # move labels to GPU if available     
    if train_on_gpu:
        labels = labels.cuda()
    # binary cross entropy with logits loss
    criterion = nn.BCEWithLogitsLoss()
    # calculate loss
    loss = criterion(D_out.squeeze(), labels)
    return loss

def fake_loss(D_out):
    '''Calculates how close discriminator outputs are to being fake.
       param, D_out: discriminator logits
       return: fake loss'''
    batch_size = D_out.size(0)
    labels = torch.zeros(batch_size)
    # move labels to GPU if available     
    if train_on_gpu:
        labels = labels.cuda()
    # binary cross entropy with logits loss
    criterion = nn.BCEWithLogitsLoss()
    # calculate loss
    loss = criterion(D_out.squeeze(), labels)
    return loss

def slerp(interpol_param, first_point, last_point):
    
    
    first_point_norm = first_point/torch.norm(first_point, dim=1, keepdim=True)
    last_point_norm = last_point/torch.norm(last_point, dim=1, keepdim=True)
    omega = torch.acos((first_point_norm*last_point_norm).sum(1))
    so = torch.sin(omega)
    
    if 0 in so:
        linear_interpol = (1.0-interpol_param) * first_point + interpol_param * last_point # L'Hopital's rule/LERP
        return linear_interpol
    
    spheric_linear_interpol = (torch.sin((1.0-interpol_param)*omega)/so).unsqueeze(1)*first_point + (torch.sin(interpol_param*omega)/so).unsqueeze(1) * last_point
    
    return spheric_linear_interpol
